      <h2>Challenges Faced by Migrant Workers</h2>

      <p>
        Migrant workers in Kerala often travel long distances in search of
        employment in construction, manufacturing, and service industries. Many
        of these workers face difficult living conditions, limited access to
        healthcare facilities, and language barriers that make communication
        with doctors difficult. Because they frequently move between workplaces
        and regions, their medical records are usually kept on paper or not
        recorded at all.
      </p>

      <p>
        This lack of accessible medical records creates serious problems when
        workers fall ill or experience accidents. Doctors may not have access to
        the worker's medical history, which can delay treatment and increase
        health risks.
      </p>

      <h3>Digital Solution for Healthcare Access</h3>

      <p>
        The Digital Health Record Management System provides a simple and

    <script>
      let records = JSON.parse(localStorage.getItem("healthRecords")) || [];

      function saveStorage() {
        localStorage.setItem("healthRecords", JSON.stringify(records));
      }

      function addRecord() {
        let name = document.getElementById("name").value;
        let age = document.getElementById("age").value;
        let location = document.getElementById("location").value;
        let condition = document.getElementById("condition").value;

        if (name == "" || age == "" || location == "" || condition == "") {
          alert("Please fill all fields");
          return;
        }

        let record = { name, age, location, condition };

        records.push(record);

        saveStorage();

        alert("Worker record saved");

        document.getElementById("name").value = "";
        document.getElementById("age").value = "";
        document.getElementById("location").value = "";
        document.getElementById("condition").value = "";
      }

      function displayRecords() {
        let container = document.getElementById("records");

        container.innerHTML = "";

        records.forEach((r, index) => {
          let div = document.createElement("div");

          div.className = "record";

          div.innerHTML = `
<b>${r.name}</b><br>
Age: ${r.age} | Location: ${r.location}<br><br>
<button onclick="viewDetails(${index})">View Details</button>
<button class="delete" onclick="deleteRecord(${index})">Delete</button>
`;

          container.appendChild(div);
        });
      }

      function deleteRecord(i) {
        records.splice(i, 1);

        saveStorage();

        displayRecords();
      }

      function searchRecord(text) {
        text = text.toLowerCase();

        let filtered = records.filter(
          (r) =>
            r.name.toLowerCase().includes(text) ||
            r.age.toString().includes(text) ||
            r.location.toLowerCase().includes(text) ||
            r.condition.toLowerCase().includes(text),
        );

        let container = document.getElementById("records");

        container.innerHTML = "";

        filtered.forEach((r, index) => {
          let div = document.createElement("div");

          div.className = "record";

          div.innerHTML = `
<b>${r.name}</b><br>
Age: ${r.age} | Location: ${r.location}<br><br>
<button onclick="viewDetails(${index})">View Details</button>
<button class="delete" onclick="deleteRecord(${index})">Delete</button>
`;

          container.appendChild(div);
        });
      }

      function viewDetails(index) {
        let worker = records[index];

        document.getElementById("recordsPage").classList.add("hidden");
        document.getElementById("detailsPage").classList.remove("hidden");

        let container = document.getElementById("workerDetails");

        container.innerHTML = `
<div class="detailsCard">
<h3>${worker.name}</h3>
<p><b>Age:</b> ${worker.age}</p>
<p><b>Work Location:</b> ${worker.location}</p>
<p><b>Health Condition:</b> ${worker.condition}</p>
<p><b>Record Type:</b> Migrant Worker Health Record</p>
<p><b>System:</b> Portable Digital Health Monitoring</p>
</div>
`;
      }

      function openRecordsPage() {
        document.getElementById("registerPage").classList.add("hidden");
        document.getElementById("detailsPage").classList.add("hidden");
        document.getElementById("recordsPage").classList.remove("hidden");

        displayRecords();
      }

      function openRegisterPage() {
        document.getElementById("recordsPage").classList.add("hidden");
        document.getElementById("registerPage").classList.remove("hidden");
      }
    </script>
