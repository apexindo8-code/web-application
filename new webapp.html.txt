          <h3>Why This System Matters</h3>

          <p>
            Migrant workers often move between workplaces and regions which
            causes loss of medical history. Without proper records doctors
            cannot easily diagnose health problems or track treatments. This
            digital platform allows clinics and healthcare providers to store
            and access worker health records quickly.
          </p>

          <h3>The Concept of Seamless Mobility</h3>

          <p>
            At its core, a portable digital health record is a patient-centric
            version of a medical file that can be accessed anytime, anywhere,
            and on any device. Unlike traditional hospital-locked systems, PDHRs
            move with you. Whether you are visiting a local clinic or seeking
            emergency care while traveling abroad, your history of allergies,
            past surgeries, and chronic conditions is available at your
            fingertips. This mobility ensures that your care is consistent,
            regardless of geographical or institutional boundaries.
          </p>

          <h3>Empowering Patient Autonomy</h3>

          <p>
            One of the most significant shifts driven by PDHRs is the move
            toward patient empowerment. Historically, doctors were the
            "gatekeepers" of medical data. With digital portability, you become
            the primary owner of your health information. Having direct access
            to lab results, imaging, and vaccination records allows you to take
            a more active role in shared decision-making. When patients can see
            their own data, they are more likely to adhere to treatment plans
            and engage in preventative care.
          </p>

    <script>
      let records = JSON.parse(localStorage.getItem("healthRecords")) || [];
      let photoBase64 = "";

      function previewPhoto(event) {
        let file = event.target.files[0];
        let reader = new FileReader();

        reader.onload = function (e) {
          photoBase64 = e.target.result;
          let img = document.getElementById("preview");
          img.src = photoBase64;
          img.classList.remove("hidden");
        };

        reader.readAsDataURL(file);
      }

      function saveStorage() {
        localStorage.setItem("healthRecords", JSON.stringify(records));
      }

      function generateID() {
        return "MW-" + Math.floor(100000 + Math.random() * 900000);
      }

      function updateStats() {
        document.getElementById("totalWorkers").innerText = records.length;
        document.getElementById("totalConditions").innerText = records.length;
      }

      function addRecord() {
        let name = document.getElementById("name").value;
        let age = document.getElementById("age").value;
        let gender = document.getElementById("gender").value;
        let contact = document.getElementById("contact").value;
        let blood = document.getElementById("blood").value;
        let location = document.getElementById("location").value;
        let hospital = document.getElementById("hospital").value;
        let doctor = document.getElementById("doctor").value;
        let condition = document.getElementById("condition").value;

        if (
          name == "" ||
          age == "" ||
          gender == "" ||
          contact == "" ||
          blood == "" ||
          location == "" ||
          hospital == "" ||
          doctor == "" ||
          condition == ""
        ) {
          alert("Please fill all fields");
          return;
        }

        let record = {
          id: generateID(),
          name,
          age,
          gender,
          contact,
          blood,
          location,
          hospital,
          doctor,
          condition,
          photo: photoBase64,
        };

        records.push(record);

        saveStorage();
        updateStats();

        alert("Worker record saved");

        location.reload();
      }

      function displayRecords() {
        let container = document.getElementById("records");
        container.innerHTML = "";

        records.forEach((r, index) => {
          let div = document.createElement("div");
          div.className = "workerCard";

          div.innerHTML = `

<div style="display:flex;align-items:center">

<img src="${r.photo}" class="workerPhoto">

<div class="workerInfo">

<h3>${r.name}</h3>

<p><b>Patient ID:</b> ${r.id}</p>
<p><b>Age:</b> ${r.age}</p>
<p><b>Gender:</b> ${r.gender}</p>
<p><b>Blood:</b> ${r.blood}</p>
<p><b>Doctor:</b> ${r.doctor}</p>

</div>

</div>

<div class="workerActions">

<button onclick="viewDetails(${index})">View</button>
<button onclick="editRecord(${index})">Edit</button>
<button class="delete" onclick="deleteRecord(${index})">Delete</button>

</div>
`;

          container.appendChild(div);
        });
      }

      function viewDetails(index) {
        let worker = records[index];

        document.getElementById("recordsPage").classList.add("hidden");
        document.getElementById("detailsPage").classList.remove("hidden");

        let container = document.getElementById("workerDetails");

        container.innerHTML = `

<div class="detailsCard">

<div class="detailsGrid">

<img src="${worker.photo}" class="detailsPhoto">

<div>

<h3>${worker.name}</h3>

<p><b>Patient ID:</b> ${worker.id}</p>
<p><b>Age:</b> ${worker.age}</p>
<p><b>Gender:</b> ${worker.gender}</p>
<p><b>Contact:</b> ${worker.contact}</p>
<p><b>Blood Group:</b> ${worker.blood}</p>
<p><b>Location:</b> ${worker.location}</p>
<p><b>Hospital:</b> ${worker.hospital}</p>
<p><b>Doctor:</b> ${worker.doctor}</p>
<p><b>Health Condition:</b> ${worker.condition}</p>

<div class="detailsBottom">

<div class="qrBox" id="qr"></div>

<button onclick="printCard('${worker.id}','${worker.name}','${worker.age}','${worker.gender}','${worker.contact}','${worker.blood}','${worker.location}','${worker.hospital}','${worker.doctor}','${worker.condition}','${worker.photo}')">
Print Health Card
</button>

</div>

</div>

</div>

</div>

`;

        let qrData = `
Patient ID:${worker.id}
Name:${worker.name}
Blood:${worker.blood}
Doctor:${worker.doctor}
`;

        new QRCode(document.getElementById("qr"), {
          text: qrData,
          width: 120,
          height: 120,
        });
      }

      function printCard(
        id,
        name,
        age,
        gender,
        contact,
        blood,
        location,
        hospital,
        doctor,
        condition,
        photo,
      ) {
        let win = window.open("", "", "width=400,height=600");

        let qrData = `ID:${id} Name:${name}`;

        win.document.write(`

<h2>Migrant Worker Health Card</h2>

<img src="${photo}" style="width:120px;border-radius:10px">

<p><b>ID:</b>${id}</p>
<p><b>Name:</b>${name}</p>
<p><b>Age:</b>${age}</p>
<p><b>Gender:</b>${gender}</p>
<p><b>Contact:</b>${contact}</p>
<p><b>Blood:</b>${blood}</p>
<p><b>Location:</b>${location}</p>
<p><b>Hospital:</b>${hospital}</p>
<p><b>Doctor:</b>${doctor}</p>
<p><b>Condition:</b>${condition}</p>

<div id="qrcode"></div>

<script src="https://cdnjs.cloudflare.com/ajax/libs/qrcodejs/1.0.0/qrcode.min.js"><\/script>

<script>

new QRCode(document.getElementById("qrcode"),{
text:"${qrData}",
width:150,
height:150
});

window.print();

<\/script>

`);
      }

      function deleteRecord(index) {
        if (confirm("Delete this record?")) {
          records.splice(index, 1);
          saveStorage();
          updateStats();
          displayRecords();
        }
      }

      function editRecord(index) {
        let r = records[index];

        let name = prompt("Edit Name", r.name);
        let age = prompt("Edit Age", r.age);
        let gender = prompt("Edit Gender", r.gender);
        let contact = prompt("Edit Contact", r.contact);
        let blood = prompt("Edit Blood Group", r.blood);
        let location = prompt("Edit Location", r.location);
        let hospital = prompt("Edit Hospital", r.hospital);
        let doctor = prompt("Edit Doctor", r.doctor);
        let condition = prompt("Edit Condition", r.condition);

        if (
          name &&
          age &&
          gender &&
          contact &&
          blood &&
          location &&
          hospital &&
          doctor &&
          condition
        ) {
          records[index] = {
            ...r,
            name,
            age,
            gender,
            contact,
            blood,
            location,
            hospital,
            doctor,
            condition,
          };

          saveStorage();
          displayRecords();
        }
      }

      function openRecordsPage() {
        document.getElementById("registerPage").classList.add("hidden");
        document.getElementById("detailsPage").classList.add("hidden");
        document.getElementById("recordsPage").classList.remove("hidden");
        displayRecords();
      }

      function openRegisterPage() {
        document.getElementById("recordsPage").classList.add("hidden");
        document.getElementById("detailsPage").classList.add("hidden");
        document.getElementById("registerPage").classList.remove("hidden");
      }

      updateStats();
